cmake_minimum_required(VERSION 3.10)
project(DungeonRPG)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add the executable
add_executable(DungeonRPG
    main.cpp
    Player.cpp
    Player.h
    Enemy.cpp
    Enemy.h
    Dungeon.cpp
    Dungeon.h
    BattleSystem.cpp
    BattleSystem.h
    Inventory.cpp
    Inventory.h
    Item.cpp
    Item.h
    GameManager.cpp
    GameManager.h
)

# Add compiler flags
if(MSVC)
    target_compile_options(DungeonRPG PRIVATE /W4)
else()
    target_compile_options(DungeonRPG PRIVATE -Wall -Wextra -Wpedantic)
endif()
